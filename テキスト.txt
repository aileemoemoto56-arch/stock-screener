// screen criteria pass/fail
function passes(c) {
  const per  = parseFloat(document.getElementById('fPER').value);
  const pbr  = parseFloat(document.getElementById('fPBR').value);
  const roe  = parseFloat(document.getElementById('fROE').value);
  const eq   = parseFloat(document.getElementById('fEQ').value);
  const sec  = document.getElementById('fSector').value;
  const stat = document.getElementById('fStatus').value;
  if (c.per > per) return false;
  if (c.pbr > pbr) return false;
  if (c.roe < roe) return false;
  if (c.eq  < eq)  return false;
  if (sec  && !c.sector.includes(sec)) return false;
  if (stat && c.status !== stat) return false;
  return true;
}

function scoreColor(v){
  if(v>=8) return 'var(--accent)';
  if(v>=6) return 'var(--green)';
  if(v>=4) return 'var(--yellow)';
  return 'var(--red)';
}
function perColor(v){ return v<=15?'var(--green)':v<=19?'var(--yellow)':'var(--red)'; }

function roeColor(v){ return v>=10?'var(--green)':v>=8?'var(--yellow)':'var(--red)'; }
function eqColor(v) { return v>=50?'var(--green)':v>=40?'var(--yellow)':'var(--red)'; }

async function fetchChartData(c) {
  const cached = chartDataCache[c.code];
  if(cached) { renderChart(c, cached); return; }

  try {
    const symbol = c.code + '.T';
    const url = `https://query1.finance.yahoo.com/v8/finance/chart/${symbol}?interval=1d&range=6mo`;
    const proxy = `https://api.allorigins.win/get?url=${encodeURIComponent(url)}`;
    const res = await fetch(proxy);
    if(!res.ok) throw new Error('通信エラー');
    const json = await res.json();
    const data = JSON.parse(json.contents);
    const result = data.chart?.result?.[0];
    if(!result) throw new Error('データなし');

    const meta = result.meta;
    const closes = result.indicators.quote[0].close;
    const timestamps = result.timestamp;
    const pairs = timestamps.map((t,i)=>({t,c:closes[i]})).filter(p=>p.c!=null);
    const prices = pairs.map(p=>p.c);

    const current = meta.regularMarketPrice || prices[prices.length-1];
    const prev    = meta.chartPreviousClose || prices[prices.length-2];
    const high52  = meta.fiftyTwoWeekHigh || Math.max(...prices);
    const low52   = meta.fiftyTwoWeekLow  || Math.min(...prices);

    function sma(arr,n){ return arr.length<n?null: arr.slice(-n).reduce((s,v)=>s+v,0)/n; }
    const d = { prices, current, prev,
      ma5:sma(prices,5), ma25:sma(prices,25),
      ma75:sma(prices,75), ma13w:sma(prices,65),
      high52, low52,
      hlPct: Math.round((current-low52)/(high52-low52)*100) };

    chartDataCache[c.code] = d;
    renderChart(c, d);
  } catch(e) {
    document.getElementById('chartSpin').style.display = 'none';
    const err = document.getElementById('chartErr');
    err.style.display = 'block';
    err.textContent = '⚠ ' + e.message + ' — データ取得に失敗しました';
  }
}

function renderChart(c, d) {
  document.getElementById('chartSpin').style.display = 'none';
  document.getElementById('chartContent').style.display = 'block';

  const fmt = v => v ? v.toLocaleString('ja-JP',{maximumFractionDigits:1}) : '—';
  const diffPct = (cur,ma) => ma ? ((cur/ma-1)*100).toFixed(1) : null;
  const diffColor = (cur,ma) => cur > ma ? 'var(--green)' : 'var(--red)';

  // MA boxes
  const maData = [
    { label:'5日移動平均',  val:d.ma5  },
    { label:'25日移動平均', val:d.ma25 },
    { label:'75日移動平均', val:d.ma75 },
    { label:'13週移動平均', val:d.ma13w },
  ];
  document.getElementById('maRow').innerHTML = maData.map(m=>{
    const dp = m.val ? diffPct(d.current,m.val) : null;
    const dc = m.val ? diffColor(d.current,m.val) : 'var(--muted)';
    return `<div class="ma-box">
      <div>
        <div class="ma-box-label">${m.label}</div>
        <div class="ma-box-val" style="color:${dc}">¥${fmt(m.val)}</div>
      </div>
      ${dp!==null?`<div class="ma-box-diff" style="color:${dc}">${d.current>m.val?'▲':'▼'}${Math.abs(dp)}%</div>`:''}
    </div>`;
  }).join('');

  // HL bar
  const hlPct = Math.max(0,Math.min(100,d.hlPct));
  document.getElementById('hlWrap').innerHTML = `
    <div style="background:var(--surface);border-radius:8px;padding:10px 12px;margin-bottom:12px;">
      <div class="hl-labels" style="margin-bottom:4px;">
        <span>年初来安値 ¥${fmt(d.low52)}</span>
        <span style="color:var(--accent);font-weight:700">現在値 ¥${fmt(d.current)}</span>
        <span>年初来高値 ¥${fmt(d.high52)}</span>
      </div>
      <div class="hl-bar-bg">
        <div class="hl-bar-fill" style="width:${hlPct}%"></div>
        <div class="hl-dot-marker" style="left:${hlPct}%"></div>
      </div>
      <div class="pos-text">高値・安値レンジの ${hlPct}% 位置</div>
    </div>`;

  // Draw canvas
  setTimeout(()=>drawCanvas(d),30);
}

function drawCanvas(d){
  const canvas = document.getElementById('mainChart');
  if(!canvas) return;
  const dpr = window.devicePixelRatio||1;
  const W = canvas.offsetWidth;
  const H = 160;
  canvas.width = W*dpr; canvas.height = H*dpr;
  const ctx = canvas.getContext('2d');
  ctx.scale(dpr,dpr);

  const prices = d.prices; const n = prices.length;
  if(n<2) return;
  const pad = {t:10,r:10,b:22,l:52};
  const cw = W-pad.l-pad.r; const ch = H-pad.t-pad.b;
  const minP = Math.min(...prices)*0.97; const maxP = Math.max(...prices)*1.03;
  const xS = i=>pad.l+(i/(n-1))*cw;
  const yS = p=>pad.t+ch-((p-minP)/(maxP-minP))*ch;

  // grid
  ctx.strokeStyle='rgba(30,45,69,0.7)'; ctx.lineWidth=0.5;
  for(let i=0;i<=4;i++){
    const y=pad.t+(i/4)*ch;
    ctx.beginPath(); ctx.moveTo(pad.l,y); ctx.lineTo(W-pad.r,y); ctx.stroke();
    const v=maxP-(i/4)*(maxP-minP);
    ctx.fillStyle='rgba(74,96,128,0.8)'; ctx.font='9px monospace';
    ctx.textAlign='right';
    ctx.fillText(Math.round(v).toLocaleString(),pad.l-4,y+3);
  }

  // MA25 dashed
  if(d.ma25 && prices.length>=25){
    ctx.strokeStyle='rgba(0,229,255,0.45)'; ctx.lineWidth=1;
    ctx.setLineDash([4,3]); ctx.beginPath();
    let st=false;
    for(let i=24;i<n;i++){
      const ma=prices.slice(i-24,i+1).reduce((s,v)=>s+v,0)/25;
      if(!st){ctx.moveTo(xS(i),yS(ma));st=true;}
      else ctx.lineTo(xS(i),yS(ma));
    }
    ctx.stroke(); ctx.setLineDash([]);
  }

  // Area fill
  const grad=ctx.createLinearGradient(0,pad.t,0,pad.t+ch);
  const up = prices[n-1]>=prices[0];
  grad.addColorStop(0, up?'rgba(0,230,118,0.15)':'rgba(255,61,90,0.12)');
  grad.addColorStop(1,'rgba(0,0,0,0)');
  ctx.fillStyle=grad; ctx.beginPath();
  ctx.moveTo(xS(0),yS(prices[0]));
  prices.forEach((p,i)=>ctx.lineTo(xS(i),yS(p)));
  ctx.lineTo(xS(n-1),pad.t+ch); ctx.lineTo(xS(0),pad.t+ch);
  ctx.closePath(); ctx.fill();

  // Line
  const lc = up?'#00e676':'#ff3d5a';
  ctx.strokeStyle=lc; ctx.lineWidth=1.8; ctx.beginPath();
  prices.forEach((p,i)=>{ if(i===0)ctx.moveTo(xS(i),yS(p)); else ctx.lineTo(xS(i),yS(p)); });
  ctx.stroke();

  // Dot
  ctx.fillStyle=lc; ctx.beginPath();
  ctx.arc(xS(n-1),yS(prices[n-1]),4,0,Math.PI*2); ctx.fill();
}

function renderFundamentals(c, body) {

    { label:'PER', val:c.per+'倍', ok:c.per<=15, crit:'15倍以下', warn:c.per<=20 },

  const allOk = criteria.filter(x=>x.ok).length;
