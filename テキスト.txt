function ec(v){return v>=50?'var(--green)':v>=40?'var(--yellow)':'var(--red)';}

function passes(c){
  var per=+document.getElementById('fPER').value;
  var pbr=+document.getElementById('fPBR').value;
  var roe=+document.getElementById('fROE').value;
  var eq =+document.getElementById('fEQ').value;
  var sec=document.getElementById('fSec').value;
  var st =document.getElementById('fStat').value;
  if(c.per>per) return false;
  if(c.pbr>pbr) return false;
  if(c.roe<roe) return false;
  if(c.eq <eq)  return false;
  if(sec && c.sector.indexOf(sec)<0) return false;
  if(st  && c.status!==st) return false;
  return true;
}

function renderCards(list){
  var g=document.getElementById('grid');
  document.getElementById('cnt').textContent=list.length;
  g.innerHTML=list.map(function(c){
    var ok=passes(c);
    var secs=c.sector.map(function(s){return '<span class="stag" style="'+SEC_C[s]+'">'+s+'</span>';}).join('');
    var topColor=ok?'background:linear-gradient(90deg,var(--green),var(--accent))':'background:var(--border)';
    return '<div class="card'+(ok?' pass':'')+'">'
      +'<div class="card-top-bar" style="'+topColor+'"></div>'
      +'<div class="card-head">'
        +'<div><div class="card-name">'+c.name+'</div><div class="card-code">'+c.code+'</div></div>'
        +'<span class="sbadge" style="'+ST_C[c.status]+'">'+c.status+'</span>'
      +'</div>'
      +'<div class="sectors">'+secs+'</div>'
      +'<div class="metrics">'
        +'<div class="met"><div class="met-l">PER</div><div class="met-v" style="color:'+pc(c.per)+'">'+c.per+'x</div></div>'
        +'<div class="met"><div class="met-l">PBR</div><div class="met-v" style="color:'+bc(c.pbr)+'">'+c.pbr+'x</div></div>'
        +'<div class="met"><div class="met-l">ROE</div><div class="met-v" style="color:'+rc(c.roe)+'">'+c.roe+'%</div></div>'
        +'<div class="met"><div class="met-l">自己資本</div><div class="met-v" style="color:'+ec(c.eq)+'">'+c.eq+'%</div></div>'
      +'</div>'

      +'<div class="cbtns">'
        +'<button class="cbtn cbtn-c" onclick="openModal(\''+c.code+'\',1)">🟢 買い戦略</button>'
        +'<button class="cbtn cbtn-s" onclick="openModal(\''+c.code+'\',2)">🔴 売り戦略</button>'

function applyFilter(){
  var list=COMPANIES.filter(passes);
  renderCards(list);
  document.getElementById('metaText').innerHTML='スクリーニング結果: <span class="cnt" id="cnt">'+list.length+'</span> 銘柄合格';
}

function resetFilter(){
  document.getElementById('fPER').value='15';
  document.getElementById('fPBR').value='3';
  document.getElementById('fROE').value='8';
  document.getElementById('fEQ').value='40';
  document.getElementById('fSec').value='';
  document.getElementById('fStat').value='';
  renderCards(COMPANIES);
  document.getElementById('metaText').innerHTML='全 <span class="cnt" id="cnt">14</span> 銘柄';
}

function switchView(v,el){
  document.querySelectorAll('.tab').forEach(function(t){t.classList.remove('on');});
  if(el) el.classList.add('on');
  document.getElementById('fStat').value = v==='watch' ? '未注目' : '';
  if(v==='pass'){
    applyFilter();
  } else if(v==='watch'){
    applyFilter();
  } else {
    resetFilter();
  }
}

// ── MODAL ──
var curCode=null, curTab=0;

function openModal(code, tab){
  curCode=code; curTab=tab||0;
  var c=COMPANIES.filter(function(x){return x.code===code;})[0];
  document.getElementById('mName').textContent=c.name;
  document.getElementById('mSub').textContent=code+' · '+c.sector.join(' / ');
  document.getElementById('ov').classList.add('on');
  showTab(curTab);
}

function closeModal(){document.getElementById('ov').classList.remove('on');}
function ovClick(e){if(e.target===document.getElementById('ov'))closeModal();}

function showTab(n){
  curTab=n;
  for(var i=0;i<4;i++){
    var t=document.getElementById('tab'+i);
    if(t) t.classList.toggle('on',i===n);
  }
  var c=COMPANIES.filter(function(x){return x.code===curCode;})[0];
  var body=document.getElementById('mbody');
  if(n===0) body.innerHTML=renderOverview(c);
  else if(n===1) body.innerHTML=renderBuy(c);
  else if(n===2) body.innerHTML=renderSell(c);
  else body.innerHTML=renderChartGuide(c);
}
